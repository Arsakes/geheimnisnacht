character_event = {
id = slaaneshcultist.1
    desc = "Your Lord demands that you do his bidding. What shall you attempt to do for him? With each task, you need to wait 1000 days for another, so as to not arouse suspicion." 
    picture = "GFX_evt_slaanesh"
    min_age = 16
    capable_only = yes
	
	is_triggered_only = yes
	
	option = {
	name = "I will donate to our cult's coffers."
	trigger = { wealth = 100 }
	wealth = -100
	random_playable_ruler = { 
	limit = { has_character_flag = hidden_slaanesh_cultist3 }

	character_event = { id = slaaneshcultist.3 }
	}
	change_variable = { which = slaaneshfavor value = 2 }  
	character_event = { id = slaaneshcultist.2 days = 1000 }	
	299 = {  change_variable = { which = slaaneshcultpower value = 1 } }
	}
	
	option = { 
	name = "Show more choices / Close Menu"
	}
	
	}
