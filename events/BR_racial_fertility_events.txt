namespace = health

##menopause for mortal women##
character_event = {
id = health.0
title = "Infertility"
desc = "You have reached the age where you can no longer have children."
picture = "GFX_evt_plague_doctor"
hide_window = yes

trigger = {
age = 45
NOT = { trait = creature_elf }
NOT = { trait = creature_dwarf }
NOT = { trait = creature_chaos_dwarf }
NOT = { culture_group = orc_group }
NOT = { culture_group = goblin_group }
}


mean_time_to_happen = {
years = 4
}

option = {
name = "OK."
add_trait = infertile
}

}


#elven infertility - each time they get pregnant, there is a chance it is their last child they can have#
character_event = {
	id = health.1
	desc = "You can feel that this will be your last child."
	picture = "GFX_evt_pregnancy"
	
		
	trigger = {
		is_pregnant = yes
		trait = creature_elf
		  }
		  
	mean_time_to_happen = {
				days = 270
			      }

	option = {
	name = "OK"
	if = {
	limit = {	
		NOT = { has_character_flag = elven_infertility }
		culture_group = asur_group
		}
          random = { 
          chance = 51
			add_trait = infertile
			set_character_flag = elven_infertility
	           }
}
	if = {
	limit = {	
		NOT = { has_character_flag = elven_infertility }
		culture_group = asrai_group
		}
          random = { 
          chance = 35
			add_trait = infertile
			set_character_flag = elven_infertility
	           }
} 
	if = {
	limit = {	
		NOT = { has_character_flag = elven_infertility }
		culture_group = druchi_group
		}
          random = { 
          chance = 35
			add_trait = infertile
			set_character_flag = elven_infertility
	           }
} 
}
}
