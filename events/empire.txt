# Ask Elector count to join emperor's wars
# FROM = Emperor 
# ROOT = Elector Count
character_event = {
    id = empire.1
    is_triggered_only = yes
    picture = "GFX_evt_emissary"
    border = GFX_event_normal_frame_war
    desc = "EVTDESC_EMPIRE_1"

    # join the emperor's wars
    option = {
	join_defender_wars = FROM
        name = "EVTOPT_EMPIRE_1A"
        FROM = { character_event = { id = empire.2 } }
        ai_chance = { factor = 0 }
    }
    option = {
        name = "EVTOPT_EMPIRE_1B"
        join_attacker_wars = FROM
        FROM = { character_event = { id = empire.2 } }
        ai_chance = { factor = 0 }
    }
    option = {
        name = "EVTOPT_EMPIRE_1C"
        join_defender_wars = FROM
        join_attacker_wars = FROM
        FROM = { character_event = { id = empire.2 } }
        ai_chance = { factor = 10 }
    }
    # not today
    option = {
        name = "EVTOPT_EMPIRE_1D"
        ai_chance = { factor = 1 }
    }
}
