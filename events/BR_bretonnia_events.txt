character_event = {
	id = bretonnia.0
	title = "Knighthood"
	desc = "You are the ruler of a Bretonnian castle, and you - like the rest of your fellow rulers - are an ordained knight of the realm."
	picture = "GFX_evt_bretonnians"
	age = 16
	
	
	trigger = {
		religion = grail
		is_ruler = yes
		is_feudal = yes
		AND = {
		NOT = { trait = bretonnian_knight }
		NOT = { trait = grail_knight }
		  }
		  }
	
		mean_time_to_happen = {
		days = 7
		}
		
	option = {
	name = "For the Lady!"
	add_trait = bretonnian_knight
	remove_trait = bretonnian_knight_errant
	}
	
	}

	age = 16

	has_landed_title = e_bretonnia
	religion = grail
	NOT = { trait = grail_knight }

	trait = bretonnian_knight
	}

	months = 1
	}

	name = "I shall quest for the grail!"

	add_trait = on_knight_quest
	set_character_flag = on_grail_quest
	character_event = { id = bretonnia.31 days = 1825 }
	}
	
	option = {
	name = "I'll abdicate..."
	trigger = { ai = no }
	abdicate = yes
	}
	
	option = {
	name = "I care nothing for what my vassals think!"

		opinion = {
			who = PREV
			modifier = opinion_suspected_heretic
			months = 360
		}
		}
		any_playable_ruler = {
			character_event = { id = bretonnia.33 days = 7 }
			}
}
